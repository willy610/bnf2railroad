s_expression  : atomic_symbol | '('  s_expression  '.'  s_expression  ')' | list ;
list  : '('  s_expression  '<'  s_expression  '>'  ')' ;
atomic_symbol  : letter  atom_part ;
atom_part  : empty | letter  atom_part | number  atom_part ;
letter  : 'a'  "..."  'z' ;
number  : '1'  "..."  '9' ;
empty  : ' ' ;
